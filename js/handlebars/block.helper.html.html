 43     // This registers a new helper method on the Handlebars template
 44     Handlebars.registerHelper("formatDate", function(date){
 45       // This guard is needed to support Blog Posts without date
 46       // the takeway point is that custom helpers parameters must be present on
 47       // the context used to render the templates
 48       // or JS error will be launched
 49       if (typeof(date) == "undefined") {
 50         return "Unknown";
 51       }
 52       // These methods need to return a String
 53       return date.getDay() + "/" + date.getMonth() + "/" + date.getFullYear();
 54     });

 56     // New helper method for Handlebars tempaltes that outputs HTML
 57     // By default handlebars escapes special chars like < or > to have your helper outputing
 58     // strings unescaped wrap the return values with Handlebars.SafeString
 59     Handlebars.registerHelper("prettyAuthor", function(author){
 60       author = author || "John Doe"
 61       return new Handlebars.SafeString("<h2>Written by " + author + " :-) </h2>");
 62     });

 69     Handlebars.registerHelper ("userScore", function (dataObject, options) {
 70       var templateWithInterpolatedData = "";
 71 
 72       for (var i = dataObject.length - 1; i >= 0; i--) {
 73         //Sum user scores from the score array and replace the array with the total
 74         dataObject[i].score = dataObject[i].score.reduce(function (
 75           prev, cur, index, array) {
 76               return prev + cur;
 77         });
 78 
 79         // Each object in the data object array is interpolated with the
 80         // options.fn method, which processes all the HTML from the template and
 81         // insert the values from the object in their respective positions.
 82 
 83         // Just so you understand the purpose of the options.fn method: it does
 84         // exactly what the regular handlebars template function does when we pass
 85         // the data object to the function, to  retrieve the values from the
 86         // object and insert them into the HTML from the template.
 87 
 88         // Without the options.fn object in this example, the raw objects
 89         // (instead of the interpolated values) would have been returned
 90 
 91         templateWithInterpolatedData += options.fn (dataObject[i]);
 92 
 93       }
 94 
 95       // We return the full string of HTML with all the values from the data
 96       // object inserted into place.
 97 
 98       return templateWithInterpolatedData;
 99     });
